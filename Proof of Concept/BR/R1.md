## BR-R1
model: 2801
```lua
en
conf t
host BR-R1
no ip domain-lookup
ipv6 unicast

!! loopback
int lo0
    ipv6 addr 2a02:a420:b:210::f:0/112
    ipv6 ospf 1 area 0
!! to ISP
int s0/1/0
    ipv6 addr 2a02:a420:b:210::1:0/127
    ipv6 ospf 1 area 0
    no sh
!! to DLS1
int fa0/0
    ipv6 addr 2a02:a420:b:210::10:0/127
    ipv6 ospf 1 area 0
    no sh

!! tunnel
interface Tunnel1
    ipv6 address 2A02:A420:B:1FF::10:1/127
    ipv6 enable
    tunnel source 2A02:A420:B:210::1:0
    tunnel mode ipsec ipv6
    tunnel destination 2A02:A420:B:110::1:0
    tunnel protection ipsec profile ipsec_profile
interface Tunnel2
    ipv6 address 2A02:A420:B:1FF::11:1/127
    ipv6 enable
    tunnel source 2A02:A420:B:210::1:0
    tunnel mode ipsec ipv6
    tunnel destination 2A02:A420:B:111::1:0
    tunnel protection ipsec profile ipsec_profile2

!! routing
ipv6 router ospf 1
    router-id 20.1.1.1
end


!! ssh apart instellen
conf t
ip domain name pjtir6.net
!! handmatig 1024 invoeren
crypto key generate rsa

ip ssh version 2
service password-encryption
username cisco password cisco
username cisco privilege 15
line vty 0 4
    login local
    transport input ssh
end

!! alleen voor de packet tracer
copy running-config startup-config
```
