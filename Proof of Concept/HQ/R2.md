## HQ-R2
model: ISR4331-SEC/K9
```lua
en
conf t
host HQ-R2
no ip domain-lookup
ipv6 unicast

!! loopback
int lo0
    ipv6 addr 2a02:a420:b:111::f:0/112
    ipv6 ospf 1 area 0
!! to ISP
int s0/1/0
    ipv6 addr 2a02:a420:b:111::1:0/127
    ipv6 ospf 1 area 0
    no sh
!! to DLS2
int g0/0/0
    ipv6 addr 2a02:a420:b:111::10:0/127
    ipv6 ospf 1 area 0
    no sh
!! to DLS1
int g0/0/1
    ipv6 addr 2a02:a420:b:111::11:0/127
    ipv6 ospf 1 area 0
    no sh

!! routing
ipv6 router ospf 1
    router-id 10.2.2.2
end


!! ssh apart instellen
conf t
ip domain name pjtir6.local
!! handmatig 1024 invoeren
crypto key generate rsa

ip ssh version 2
service password-encryption
username cisco password cisco
username cisco privilege 15
line vty 0 4
    login local
    transport input ssh
end

!! alleen voor de packet tracer
copy running-config startup-config
```
